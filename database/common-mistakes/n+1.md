```sql
-- First, fetch all students:
SELECT * FROM students;

-- Then, for each student, fetch their courses one by one:
SELECT * FROM courses WHERE student_id = ?;

-- for 100 students, total 101 queries (100 queries for courses + 1 query for students) -> N + 1
```

```sql
-- join approach
SELECT students.*, courses.*
FROM students
LEFT JOIN courses ON students.id = courses.student_id;

-- batch query approach
SELECT * FROM students;
SELECT * FROM courses WHERE student_id IN (1, 2, 3, 4, 5);
```
